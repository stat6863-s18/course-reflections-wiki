# Recipes Exploration Results



```


*Example for Approach1;

proc sql;
    select
        Species
        ,SepalLength
        ,SepalWidth
        ,PetalLength
        ,PetalWidth
    from
        sashelp.iris
    ;
quit;

*Example for Approach2;

proc contents order = varnum data = sashelp.iris;
run;

proc sql;
    select
        Species
        ,SepalLength
        ,SepalWidth
        ,PetalLength
        ,PetalWidth
    from
        sashelp.iris
    ;
quit;

*Example for Approach3;

proc sql;
    select
        name
    from
        dictionary.columns
    where
        lowcase(libname) = ‘sashelp’
    and
        lowcase(memname) = ‘iris’
    ;
quit;

proc sql;
    select
        Species
        ,SepalLength
        ,SepalWidth
        ,PetalLength
        ,PetalWidth
    from
        sashelp.iris
    ;
quit;
```
